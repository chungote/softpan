CREATE TABLE Ordenes_Compra(
        nro_oc int NOT NULL primary key ,
        abierta smallint NOT NULL DEFAULT 0,
        fecha datetime NOT NULL,
        aprobado_a smallint NOT NULL DEFAULT 0,
        fecha_aprobacion datetime NULL,
        cod_proveedor int NOT NULL,
        cod_moneda smallint NOT NULL,
        tipo_comprobante smallint NOT NULL,
        cod_condicion smallint NOT NULL,
        cod_plazo smallint NOT NULL,
        fecha_entrega datetime NOT NULL,
        recibido smallint NOT NULL DEFAULT 0,
        impreso smallint NOT NULL DEFAULT 0,
        observacion varchar(100) NULL,
        Estado varchar(1)  NOT NULL DEFAULT 'G',
        cod_comprador smallint NOT NULL DEFAULT 0,
        dias smallint NOT NULL DEFAULT 0,
        importe decimal(18,0),
        fecha_alta date ,
        problema int NOT NULL DEFAULT 0,
        observacion_prob varchar(300)  NULL)
